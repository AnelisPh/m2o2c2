	\begin{abstract}
		Second order partial derivatives are partial derivatives of partial derivatives
	\end{abstract}
	
	\begin{definition}
		If $f:\R^n \to \R$ is a function, then $\frac{\partial f}{\partial x_i}:\R^n \to \R$ is another function, so we can take its partial derivative with respect to $x_j$.
		We define the \texbf{second order partial derivative} $\frac{\partial^2 f}{\partial x_j \partial x_i} = \frac{\partial}{\partial x_j} \frac{\partial f}{\partial x_i}$.  In the special case
		that $i=j$ we will write $\frac{\partial^2 f}{\partial x_i^2}$ (even though this notation is horrible, it is standard, so we will follow it).
	\end{definition}
	
	\begin{question}
		Let $f(x,y) = x^2y^3$
		\begin{solution}
			\begin{hint}
				\begin{align*}
				\frac{\partial^2f}{\partial x^2} &= \frac{\partial }{\partial x}\left( \frac{\partial }{\partial x} (x^2y^3)\right)\\
					&=\frac{\partial }{\partial x} (2xy^3)\\
					&=2y^3
				\end{align*}
			\end{hint}
			$ \frac{\partial^2f}{\partial x^2} =$ \answer{$2y^3$}
		\end{solution}
		\begin{solution}
		\begin{hint}
				\begin{align*}
				\frac{\partial^2f}{\partial x \partial y} &= \frac{\partial }{\partial x}\left( \frac{\partial }{\partial y} (x^2y^3)\right)\\
					&=\frac{\partial }{\partial x} (3x^2y^2)\\
					&=6xy^2
				\end{align*}
			\end{hint}
			$\frac{\partial^2f}{\partial x \partial y} =$ \answer{$6xy^2$}
		\end{solution}
		\begin{solution}
		\begin{hint}
				\begin{align*}
				\frac{\partial^2f}{\partial y \partial x} &= \frac{\partial }{\partial y}\left( \frac{\partial }{\partial x} (x^2y^3)\right)\\
					&=\frac{\partial }{\partial y} (2xy^3)\\
					&=6xy^2
				\end{align*}
			\end{hint}
			$\frac{\partial^2f}{\partial y \partial x} =$ \answer{$6xy^2$}
		\end{solution}
		\begin{solution}
		\begin{hint}
				\begin{align*}
				\frac{\partial^2f}{\partial y^2} &= \frac{\partial }{\partial y}\left( \frac{\partial }{\partial y} (x^2y^3)\right)\\
					&=\frac{\partial }{\partial y} (3x^2y^2)\\
					&=6x^2y
				\end{align*}
			\end{hint}
			$\frac{\partial^2f}{\partial y^2} =$ \answer{$6x^2y$}
		\end{solution}
		\begin{solution}
		Did you notice how $\frac{\partial^2f }{\partial x\partial y} = \frac{\partial^2f }{\partial y\partial x}$?  Doesn't that fill you with a sense of wonder and mystery?
		\begin{multiple-choice}
			\choice[correct]{Yes!}
			\choice{No :(}
		\end{multiple-choice}
		\end{solution}
	\end{question}
	
	\begin{question}
		Let $f(x,y) = \sin(xy^2)$
		\begin{solution}
			\begin{hint}
				\begin{align*}
				\frac{\partial^2f}{\partial x^2} &= \frac{\partial }{\partial x}\left( \frac{\partial }{\partial x} \sin(xy^2)\right)\\
					&=\frac{\partial }{\partial x} (y^2\sin(xy^2))\\
					&=y^4\sin(xy^2)
				\end{align*}
			\end{hint}
			$ \frac{\partial^2f}{\partial x^2} =$ \answer{$y^4sin(xy^2)$}
		\end{solution}
		\begin{solution}
		\begin{hint}
				\begin{align*}
				\frac{\partial^2f}{\partial x \partial y} &= \frac{\partial }{\partial x}\left( \frac{\partial }{\partial y} (\sin(xy^2))\right)\\
					&=\frac{\partial }{\partial x} (2xy\cos(xy^2))\\
					&=2y\cos(xy^2)-2xy(y^2)\sin(xy^2)\\
					&=2y\cos(xy^2)-2xy^3\sin(xy^2)
				\end{align*}
			\end{hint}
			$\frac{\partial^2f}{\partial x \partial y} =$ \answer{$2ycos(xy^2)-2xy^3sin(xy^2)$}
		\end{solution}
		\begin{solution}
		\begin{hint}
				\begin{align*}
				\frac{\partial^2f}{\partial y \partial x} &= \frac{\partial }{\partial y}\left( \frac{\partial }{\partial x} (\sin(xy^2))\right)\\
					&=\frac{\partial }{\partial y} (y^2\cos(xy^2))\\
					&=2y\cos(xy^2)-y^2(2xy)\sin(xy^2)\\
					&=2y\cos(xy^2)-2xy^3\sin(xy^2)
				\end{align*}
			\end{hint}
			$\frac{\partial^2f}{\partial y \partial x} =$ \answer{$2ycos(xy^2)-2xy^3sin(xy^2)$}
		\end{solution}
		\begin{solution}
		\begin{hint}
				\begin{align*}
				\frac{\partial^2f}{\partial y^2} &= \frac{\partial }{\partial y}\left( \frac{\partial }{\partial y} (\sin(xy^2))\right)\\
					&=\frac{\partial }{\partial y} (2xy\cos(xy^2))\\
					&=2x\cos(xy^2)-2xy(2xy)\sin(xy^2)\\
					&=2x\cos(xy^2)-4x^2y^2\sin(xy^2)
				\end{align*}
			\end{hint}
			$\frac{\partial^2f}{\partial y^2} =$ \answer{$2xcos(xy^2)-4x^2y^2sin(xy^2)$}
		\end{solution}
		
		This lends even more evidence to the startling claim that $\frac{\partial^2 f}{\partial x \partial y} = \frac{\partial^2 f}{\partial y \partial x}$
	\end{question}
	
	\begin{question}
		Let $f:\R^3 \to \R$ be the function $f(x,y,z) = x^2yz^3$
		\begin{solution}
			\begin{hint}
				\begin{align*}
					\frac{\partial^2 f}{\partial x \partial z} &= \frac{\partial }{ \partial x} \frac{\partial}{ \partial z} x^2yz^3\\
						&=\frac{\partial }{ \partial x} 3x^2yz^2\\
						&= 6xyz^2
				\end{align*}
			\end{hint}
			$\frac{\partial^2 f}{\partial x \partial z} = $\answer{6xyz^2}
		\end{solution}
		\begin{solution}
		\begin{hint}
				\begin{align*}
					\frac{\partial^2 f}{\partial z \partial x} &= \frac{\partial }{ \partial z} \frac{\partial}{ \partial x} x^2yz^3\\
						&=\frac{\partial }{ \partial x} 2xyz^3\\
						&= 6xyz^2
				\end{align*}
			\end{hint}
			$\frac{\partial^2 f}{\partial z \partial x} = $\answer{6xyz^2}
		\end{solution}
		
		Okay, it really really looks like $\frac{\partial^2 f}{\partial x_i \partial x_j} = \frac{\partial^2 f}{\partial x_j \partial x_i} $.
		\end{question}
