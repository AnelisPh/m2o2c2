\title{Increasing and decreasing}

\begin{document}
	\begin{abstract}
		Consider whether a function is increasing or decreasing.
	\end{abstract}\maketitle
	
	\begin{question}
		Let $f:\R^2 \to \R$ be given by $f(x,y)=(x+2y)^2+y^3$
		\begin{solution}
			\begin{hint}
				By computing the directional derivative in that direction, we see that $f$ is decreasing. 
			\end{hint}
			At the point $(-1,5)$, is $f$ increasing or decreasing in the direction $\verticalvector{1\\-1}$.
			\begin{multiple-choice}
				\choice{Increasing}
				\choice[correct]{Decreasing}
			\end{multiple-choice}
		\end{solution}
	\end{question}
	
	\begin{question}
		Let $f:\R^3 \to \R$ be given by $f(x,y,z)=\frac{x+y}{z}$
		\begin{solution}
			\begin{hint}
				By computing the directional derivative in that direction, we see that $f$ is decreasing. 
				You could also see that $f$ is decreasing by noting that the numerator is left unchanged when $x$ and $y$ are increased
				equally in opposite directions, but the denominator is increasing.
			\end{hint}
			At the point $(3,2,6)$, is $f$ increasing or decreasing in the direction $\verticalvector{1\\-1\\2}$.
			\begin{multiple-choice}
				\choice{Increasing}
				\choice[correct]{Decreasing}
			\end{multiple-choice}
		\end{solution}
	\end{question}
	
	\begin{question}
		Let $f:\R^4 \to \R$ be given by $f(x,y,z,t)=\frac{xy^2}{t}+z^2$
		\begin{solution}
			\begin{hint}
				By computing the directional derivative in that direction, we see that $f$ is decreasing.
			\end{hint}
			At the point $(0,1,5,-1)$, is $f$ increasing or decreasing in the direction $\verticalvector{1\\1\\-2\\3}$.
			\begin{multiple-choice}
				\choice{Increasing}
				\choice[correct]{Decreasing}
			\end{multiple-choice}
		\end{solution}
	\end{question}
